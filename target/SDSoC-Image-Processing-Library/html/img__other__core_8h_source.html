525  }
526  } else {

528  bl = br = data_old;
529  tl = tr = data_left;

548

576  STATIC_ASSERT(SCALE_TYPE == VX_INTERPOLATION_NEAREST_NEIGHBOR || SCALE_TYPE == VX_INTERPOLATION_BILINEAR, scale_type_not_supported);
577

579  if (SCALE_TYPE == VX_INTERPOLATION_NEAREST_NEIGHBOR) {

595 template<const vx_uint16 IMG_COLS, const vx_uint16 IMG_ROWS>
596 void integral(vx_uint8* input, vx_uint32* output) {
597 #pragma HLS INLINE
598 
599  // Buffers 1 row of the integral image
600  vx_uint32 buffer[IMG_COLS];
601 
602  // Compute integral on image
603  for (vx_uint32 y = 0; y < IMG_ROWS; y++) {
604  for (vx_uint32 x = 0, sum_row = 0; x < IMG_COLS; x++) {
605 #pragma HLS PIPELINE II=1
606 
607  // Read input
608  vx_uint8 pixel = input[y*IMG_COLS + x];
609 
610  // Compute the sum of the current row
611  sum_row += +pixel;
612 
613  // Compute the integral pixel
614  vx_uint32 sum_area = sum_row;
615  if (y > 0)
616  sum_area += buffer[x];
617 
618  // Buffer the integral pixels of row for next row
619  buffer[x] = sum_area;
620 
621  // Write output
622  output[y*IMG_COLS + x] = sum_area;
623  }
624  }

632 template<typename IN_TYPE, const vx_uint32 IMG_PIXELS, const vx_uint32 DISTRIBUTION_BINS, const vx_uint32 DISTRIBUTION_RANG, const IN_TYPE DISTRIBUTION_OFFSET>
633 void histogram(IN_TYPE input[IMG_PIXELS], vx_uint32 output[IMG_PIXELS]) {
634 #pragma HLS INLINE
635 
636  // Check Datatype
637  const vx_type_e TYPE = GET_TYPE(IN_TYPE);
638  const bool allowed_type = (TYPE == VX_TYPE_UINT8) || (TYPE == VX_TYPE_UINT16);
639  STATIC_ASSERT(allowed_type, data_type_is_not_allowed_for_histogram);
640 
641  // Compute Scale (BINS / RANGE)
642  const vx_uint32 ACCURACY = 12;
643  const vx_uint32 SCALE = static_cast<vx_uint32>((static_cast<double>(DISTRIBUTION_BINS) / static_cast<double>(DISTRIBUTION_RANG)) * pow(2, ACCURACY));
644 
645  // Check for overflow
646  const vx_uint32 MAX_DATA_RANGE = ACCURACY + (DISTRIBUTION_BINS / DISTRIBUTION_RANG) + 1 + sizeof(IN_TYPE);
647  STATIC_ASSERT(MAX_DATA_RANGE <= 32, overflow__reduce_bins_or_increase_range);
648 
649  //Global variables
650  vx_uint32 pre_bin = 0, cur_bin = 0;
651  vx_uint32 pre_val = 0, cur_val = 0;
652 
653  // Buffer Histogram in BRAM
654  vx_uint32 LUTA[DISTRIBUTION_BINS];
655  vx_uint32 LUTB[DISTRIBUTION_BINS];
656 
657  //filling with zeros
658  for (vx_uint32 i = 0; i < DISTRIBUTION_BINS; i++) {
659 #pragma HLS PIPELINE II=1
660  LUTA[i] = 0;
661  LUTB[i] = 0;
662  }
663 
664  // Compute Histogram
665  for (vx_uint32 i = 0; i < IMG_PIXELS + 1; i++) {
666 #pragma HLS PIPELINE II=1
667 
668  // Gett the input data
669  IN_TYPE pixel = 0;
670  if (i < IMG_PIXELS)
671  pixel = input[i];
672 
673  // Compute the BIN
674  vx_uint16 index = (vx_uint16)pixel - (vx_uint16)DISTRIBUTION_OFFSET;
675  if (DISTRIBUTION_BINS != DISTRIBUTION_RANG)
676  cur_bin = ((vx_uint32)index * SCALE) >> ACCURACY;
677  else
678  cur_bin = (vx_uint32)index;
679 
680  // Read current bin & write previous bin
681  if (i % 2 == 0) {
682  cur_val = LUTA[cur_bin];
683  LUTB[pre_bin] = pre_val;
684  } else {
685  cur_val = LUTB[cur_bin];
686  LUTA[pre_bin] = pre_val;
687  }
688 
689  // Update to store in next iteration
690  cur_val++;
691  pre_bin = cur_bin;
692  pre_val = cur_val;
693  }
694 
695  // Write output data
696  for (vx_uint32 i = 0; i < DISTRIBUTION_BINS; i++) {
697 #pragma HLS PIPELINE II = 1
698  output[i] = LUTA[i] + LUTB[i];
699  }
700 }

703 template<typename DATA_TYPE, const vx_uint32 IMG_PIXELS, const vx_uint32 LUT_COUNT, const vx_uint32 LUT_OFFSET>
704 void tableLookup(DATA_TYPE input[IMG_PIXELS], DATA_TYPE lut[LUT_COUNT], DATA_TYPE output[IMG_PIXELS]) {
705 #pragma HLS INLINE
706 
707  // Check Datatype
708  const vx_type_e TYPE = GET_TYPE(DATA_TYPE);
709  const bool allowed_type = (TYPE == VX_TYPE_UINT8) || (TYPE == VX_TYPE_INT16);
710  STATIC_ASSERT(allowed_type, data_type_is_not_allowed_for_histogram);
711 
712  // Buffer for lookup table
713  DATA_TYPE table[LUT_COUNT];
714 
715  // Read lookup table into buffer
716  for (vx_uint32 i = 0; i < LUT_COUNT; i++) {
717 #pragma HLS PIPELINE II=1
718  table[i] = lut[i];
719  }
720 
721  // Perform table lookup
722  for (vx_uint32 i = 0; i < IMG_PIXELS; i++) {
723 #pragma HLS PIPELINE II=1
724 
725  // Read from Input
726  DATA_TYPE src = input[i];
727 
728  // Add Offset
729  vx_int32 index = (vx_int32)src + (vx_int32)LUT_OFFSET;
730 
731  // Perform Table Lookup and write to output
732  if ((index >= 0) && (index < (vx_int32)LUT_COUNT))
733  output[i] = table[index];
734  }
735 }
