201 void swTestGaussian(const float * input, float * output, const vx_int32 rows, const vx_int32 cols) {
202 
203  // Constants
204  const vx_uint16 KERN_RAD = KERN_SIZE / 2;
205  const float SIGM = static_cast<float>(SIGMA) / 256.0f;
206  const float s = 2.0f * SIGM * SIGM;
207 
208  // Variables
209  float sum = 0.0f;
210  float gaussKernel[KERN_SIZE][KERN_SIZE];
211 
212  // Generate gaussian kernel
213  for (vx_uint16 x = 0; x < KERN_SIZE; x++) {
214 #pragma HLS unroll
215  for (vx_uint16 y = 0; y < KERN_SIZE; y++) {
216 #pragma HLS unroll
217  const vx_uint32 a = static_cast<vx_uint32>(x - KERN_RAD);
218  const vx_uint32 b = static_cast<vx_uint32>(y - KERN_RAD);
219  const float c = static_cast<float>(a*a + b*b);
220  const float r = sqrtf(c);
221  const float result = (expf(-(r*r) / s)) / (static_cast<float>(M_PI) * s);
222  gaussKernel[x][y] = result;
223  sum += result;
224  }
225  }
226 
227  // Normalize gaussian kernel
228  for (vx_uint16 x = 0; x < KERN_SIZE; ++x) {
229 #pragma HLS unroll
230  for (vx_uint16 y = 0; y < KERN_SIZE; ++y) {
231 #pragma HLS unroll
232  gaussKernel[x][y] /= sum;
233  }
234  }
235 
236  // Compute gaussian filter
237  for (vx_int32 y = 0; y < rows; y++) {
238  for (vx_int32 x = 0; x < cols; x++) {
239 
240  // Data is passed through if filter exceeds border
241  if (BORDER_TYPE == VX_BORDER_UNDEFINED) {
242  if (x < KERN_RAD || x >= cols - KERN_RAD || y < KERN_RAD || y >= rows - KERN_RAD) {
243  output[y*cols + x] = input[y*cols + x];
244  continue;
245  }
246  }
247 
248  // Compute single gaussian filter
249  sum = 0.0f;
250  for (vx_int32 i = -KERN_RAD; i <= KERN_RAD; i++) {
251  for (vx_int32 j = -KERN_RAD; j <= KERN_RAD; j++) {
252 
253  // Replicated border
254  if (BORDER_TYPE == VX_BORDER_REPLICATE) {
255  vx_int32 yi = min(max(y + i, static_cast<vx_int32>(0)), rows - 1);
