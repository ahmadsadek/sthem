54 template<typename InType, typename CompType, const InType MAXIMUM, const bool CHECK_MAX, const vx_round_policy_e ROUND_POLICY>
55 InType sqrtLester(CompType value) {
56 #pragma HLS INLINE
57 
58  // Number of stages (latency)
59  const vx_uint8 N = sizeof(InType) * 8;
60 
61  // Variables
62  InType A1 = 0; // A^1 Intermediate result
63  CompType A2 = 0; // A^2 Square of the intermediate result
64 
65  // Each stage computes 1 bit of the resulting vector
66  for (vx_uint8 n = N - 1; n < N; n--) {
67 #pragma HLS unroll
68 
69  // Add new bit of position n and compute (A1 + B1)^2
70  InType B1 = static_cast< InType>(1) << (n);
71  CompType B2 = static_cast<CompType>(B1) << (n);
72  CompType AB = static_cast<CompType>(A1) << (n);
73  CompType A2_next = A2 + B2 + (AB << 1); // A*A + B*B + 2*A*B
74 
75  // Alternative -> more LUT, less FF
76  // CompType A2_next = A2 + (static_cast<CompType>(B1 | (A1 << 1)) << n);
77 
78  // Store if tmp does not exceed value
79  if (A2_next <= value) {
80  A1 |= B1;
81  A2 = A2_next;
82  }
83  }
84 
85  // Round to the nearest integer and check for overflow
86  if (ROUND_POLICY == VX_ROUND_POLICY_TO_NEAREST_EVEN) {
87  if (CHECK_MAX) {
88  if (((value - A2) > static_cast<CompType>(A1)) && (A1 != MAXIMUM))
89  A1++;
90  } else {
91  if ((value - A2) > static_cast<CompType>(A1))
92  A1++;
93  }
94  }
95 
96  // Return result
97  return A1;
98 }

410 template<typename ScalarType, typename CompType, typename NormType, const bool IS_SIGNED, const vx_uint16 TYPE_SIZE, const vx_convert_policy_e CONV_POLICY, const vx_round_policy_e ROUND_POLICY, const vx_uint16 SCALE, const CompType MAX_VAL, const CompType MIN_VAL>
411 ScalarType multiply(ScalarType inScalarA, ScalarType inScalarB) {
412 #pragma HLS INLINE
413 
414  // Store result
415  CompType D = 0, result = 0;
416 
417  // Check if scaling is by a power of two
418  const bool IS_POWER_TWO = (SCALE & (SCALE - 1)) == 0;
419 
420  // Compute the amount of bits data needs to be shiftet, if it is power of two
421  vx_uint16 SHIFT = 0;
422  for (vx_uint16 i = 0; i < sizeof(vx_uint16) * 8; i++) {
423 #pragma HLS unroll
424  if ((SCALE & static_cast<vx_uint16>(1 << i)) != 0)
425  SHIFT = i;
426  }
427 
428  // Compute the scale, if it is not a power of two
429  int e = 0;
430  const double normalization = 1.0 / static_cast<double>(max(SCALE, static_cast<vx_uint16>(1)));
431  frexp(normalization, &e);
432  vx_uint32 fractionA = 1, fractionB = 1;
433  for (vx_int32 i = 0; i < 24; i++) {
434 #pragma HLS unroll
435  fractionA *= 2;
436  }
437  for (vx_int32 i = 0; i < (-e); i++) {
438 #pragma HLS unroll
439  fractionB *= 2;
440  }
441  const vx_int16 MULT_SHIFT = static_cast<vx_int16>(24 - e);
442  const NormType MULT_INT = static_cast<NormType>(normalization * static_cast<double>(fractionA)* static_cast<double>(fractionB) + 0.5);
443  const double MULT_DBL = 1.0 / static_cast<double>(max(SCALE, static_cast<vx_uint16>(1)));
444 
445 
446  // Scale should not be negative, scale of zero gives no result
447  if (SCALE > 0) {
448 
449  // Compute Multiplicaion
450  CompType A = static_cast<CompType>(inScalarA);
451  CompType B = static_cast<CompType>(inScalarB);
452  CompType C = A*B;
453 
454  // Multiply and truncate to nearest integer
455  if (ROUND_POLICY == VX_ROUND_POLICY_TO_NEAREST_EVEN) {
456 
457  // Shift bits, if scale it is a power of two
458  if (IS_POWER_TWO) {
459  D = (C + (static_cast<CompType>(1) << (SHIFT - 1))) >> SHIFT;
460 
461  // Multiply by scale, if it is not a power of two
462  } else {
463 
464  // Use floating point multiplication if data size is 32 bit
465  if (TYPE_SIZE == 4) {
466  D = static_cast<CompType>(static_cast<double>(C)* MULT_DBL + 0.5);
467 
468  // Use 24 bit precise integer multiplication if data size is 16 or 8 bit
469  } else {
470  D = static_cast<CompType>((static_cast<NormType>(C)* MULT_INT + (static_cast<NormType>(1) << (MULT_SHIFT - 1))) >> MULT_SHIFT);
471  }
472  }
473 
474  // Multiply and truncate the least significant values
475  } else if (ROUND_POLICY == VX_ROUND_POLICY_TO_ZERO) {
476 
477  // Shift bits, if scale it is a power of two
478  if (IS_POWER_TWO) {
479  D = (C) >> SHIFT;
480 
481  // Multiply by scale, if it is not a power of two
482  } else {
483 
484  // Use floating point multiplication if data size is 32 bit
485  if (TYPE_SIZE == 4) {
486  D = static_cast<CompType>(static_cast<double>(C)* MULT_DBL);
487 
488  // Use 24 bit precise integer multiplication if data size is 16 or 8 bit
489  } else {
490  D = static_cast<CompType>((static_cast<NormType>(C)* MULT_INT) >> MULT_SHIFT);
491  }
492  }
493  }
494 
495  // Saturate the result
496  if (CONV_POLICY == VX_CONVERT_POLICY_SATURATE) {
497  if (IS_SIGNED) {
498  result = max(min(D, MAX_VAL), MIN_VAL);
499  } else {
500  result = min(D, MAX_VAL);
501  }
502 
503  // Take LSBs
504  } else if (CONV_POLICY == VX_CONVERT_POLICY_WRAP) {
505  result = D;
506  }
507  }
508 
509  // Return result
510  return static_cast<ScalarType>(result);
511 }

603  result = computeAbsDiff<ScalarType, IS_SIGNED, TYPE_SIZE, VX_CONVERT_POLICY_SATURATE>(inScalarA[j], inScalarB[j]);
604

606  } else if (OPERATION_TYPE == SDSOC::ARITHMETIC_ADDITION) {
607  result = computeAddition<ScalarType, IS_SIGNED, TYPE_SIZE, CONV_POLICY>(inScalarA[j], inScalarB[j]);
608

610  } else if (OPERATION_TYPE == SDSOC::ARITHMETIC_SUBTRACTION) {

614  } else if (OPERATION_TYPE == SDSOC::MAGNITUDE) {

618  } else if (OPERATION_TYPE == SDSOC::MULTIPLY) {

624  }

628

631  }

647 template<typename ScalarType, typename VecType, const vx_uint32 IMG_PIXEL, SDSOC::bitwiseOperation OPERATION_TYPE>
648 void operationBitwise(VecType *in1, VecType *in2, VecType *out) {
649 #pragma HLS INLINE
650 
651  // Constants
652  const vx_uint16 VEC_SIZE = static_cast<vx_uint16>(sizeof(VecType) / sizeof(ScalarType));
653  const vx_uint32 VEC_PIXEL = IMG_PIXEL / static_cast<vx_uint32>(VEC_SIZE);
654 
655  // Check function parameters/types
656  checkDataTypeOperation<ScalarType, VecType, IMG_PIXEL>();
657  STATIC_ASSERT((numeric_limits<ScalarType>::is_signed == false), data_type_must_be_unsigned);
658 
659  // Computes bitwise operations (pipelined)
660  for (vx_uint32 i = 0; i < VEC_PIXEL; i++) {
661 #pragma HLS PIPELINE II=1
662 
663  // Scalar arrays
664  ScalarType inScalarA[VEC_SIZE];
665 #pragma HLS array_partition variable=inScalarA complete dim=0
666  ScalarType inScalarB[VEC_SIZE];
667 #pragma HLS array_partition variable=inScalarB complete dim=0
668  ScalarType outScalar[VEC_SIZE];
669 #pragma HLS array_partition variable=outScalar complete dim=0
670 
671  // Vector variables
672  VecType inVectorA = in1[i];
673  VecType inVectorB = 0;
674  if (OPERATION_TYPE != SDSOC::BITWISE_NOT)
675  inVectorB = in2[i];
676  VecType outVector = 0;
677 
678  // Converts data from vector to scalar
679  convertVectorToScalar<VecType, ScalarType, 1, VEC_SIZE>(&inVectorA, &inScalarA);
680  if (OPERATION_TYPE != SDSOC::BITWISE_NOT)
681  convertVectorToScalar<VecType, ScalarType, 1, VEC_SIZE>(&inVectorB, &inScalarB);
682 
683  // Compute a vector of bitwise operations in 1 clock cycle
684  for (vx_uint16 j = 0; j < VEC_SIZE; j++){
685 #pragma HLS unroll
686 
687  // Performs a bitwise AND operation between two unsigned images.
688  if (OPERATION_TYPE == SDSOC::BITWISE_AND) {
689  outScalar[j] = inScalarA[j] & inScalarB[j];
690 
691  // Performs a bitwise EXCLUSIVE OR (XOR) operation between two unsigned images.
692  } else if (OPERATION_TYPE == SDSOC::BITWISE_XOR) {
693  outScalar[j] = inScalarA[j] ^ inScalarB[j];
694 
695  // Performs a bitwise INCLUSIVE OR operation between two unsigned images.
696  } else if (OPERATION_TYPE == SDSOC::BITWISE_OR) {
697  outScalar[j] = inScalarA[j] | inScalarB[j];
698 
699  // Performs a bitwise NOT operation on a input images.
700  } else if (OPERATION_TYPE == SDSOC::BITWISE_NOT) {
701  outScalar[j] = ~(inScalarA[j]);
702  }
703  }
704 
705  // Convert data back to vector type
706  convertScalarToVector<VecType, ScalarType, 1, VEC_SIZE>(&outScalar, &outVector);
707 
708  // Write result into global memory
709  out[i] = outVector;
710  }
711 }
